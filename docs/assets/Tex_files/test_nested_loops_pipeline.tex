    %% create tasks %%
    %%%%%%%%%%%%%%%%%%
    %

    %        options                                id    coords label                          sin sout sfwd attributes

    \task                                           {t1}  {0, 0} {initialize\\($t_1$)}          {0} {1}  {0}  {"seq"};
